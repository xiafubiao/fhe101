## Installation

See: [TFHE-rs Installation](https://docs.zama.ai/tfhe-rs/get-started/installation)

```bash
# Install Rust
curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh
source ~/.cargo/env

# Create project
cargo new tfhe-example
cd tfhe-example

# Add dependency
echo 'tfhe = { version = "~1.6.1", features = ["integer"] }' >> Cargo.toml

# Run
cargo run --release
```

## Official Example Code

Source: [TFHE-rs Quick Start](https://docs.zama.ai/tfhe-rs/get-started/quick-start)

```rust
use tfhe::{ConfigBuilder, generate_keys, set_server_key, FheUint8};
use tfhe::prelude::*;

fn main() {
    let config = ConfigBuilder::default().build();

    // Client-side: generate keys
    let (client_key, server_key) = generate_keys(config);

    let clear_a = 27u8;
    let clear_b = 128u8;

    // Client-side: encrypt data
    let a = FheUint8::encrypt(clear_a, &client_key);
    let b = FheUint8::encrypt(clear_b, &client_key);

    // Server-side: compute on ciphertext
    set_server_key(server_key);
    let result = a + b;

    // Client-side: decrypt result
    let decrypted_result: u8 = result.decrypt(&client_key);

    let clear_result = clear_a + clear_b;
    assert_eq!(decrypted_result, clear_result);
}
```
