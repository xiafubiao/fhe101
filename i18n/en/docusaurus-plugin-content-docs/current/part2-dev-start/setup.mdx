## Quick Start: Hello FHE

Using TFHE-rs as an example, experience homomorphic encryption in 5 minutes:

```rust
use tfhe::prelude::*;
use tfhe::{generate_keys, set_server_key, ConfigBuilder};

fn main() {
    // 1. Configure parameters
    let config = ConfigBuilder::all_disabled()
        .enable_default_integers()
        .build();

    // 2. Generate keys
    let (client_key, server_key) = generate_keys(config);

    // 3. Encrypt on client
    set_server_key(server_key);
    let a = FheInt32::encrypt(42, &client_key);
    let b = FheInt32::encrypt(1, &client_key);

    // 4. Compute on ciphertext in the cloud (doesn't touch plaintext)
    let result = &a + &b;

    // 5. Client decrypts
    let decrypted: i32 = result.decrypt(&client_key);
    println!("42 + 1 = {}", decrypted); // Outputs 43
}
```
